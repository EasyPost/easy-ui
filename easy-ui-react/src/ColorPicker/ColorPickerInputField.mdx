# ColorPickerInputField

A `<ColorPickerInputField />` allows users to select a color with a color picker or as a text input.

<Canvas of={ColorPickerInputFieldStories.Default} />

## Rendering

A `<ColorPickerInputField />` is a standalone component that matches the API of Easy UI's other `InputField`s.

```tsx
import { ColorPickerInputField, Color } from "@easypost/easy-ui/ColorPicker";

function StandaloneColorPickerInputField() {
  const [color, setColor] = React.useState<Color | null>(null);

  // Use the selected color
  console.log(color.toString("css"));

  return (
    <ColorPickerInputField
      label="Primary color"
      value={color}
      onChange={setColor}
      helperText="Select a color"
      validationState="valid"
      errorText="Invalid color"
      isDisabled={false}
      isReadOnly={false}
      isRequired={false}
      autoFocus
    />
  );
}
```
